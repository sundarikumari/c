// Online C compiler to run C program online
#include <stdio.h>

void main() {
    int i,j;
    char ch='A';
    for(i=1;i<=5;i++)
    {
        for(j=1;j<=i;j++)
        {
            printf("%2c",ch);
            ch++;
        }
        printf("\n");
    }
}
